```svelte
<script lang="ts">
  import * as editable from "@ibrains-design/editable"
  import { useMachine, normalizeProps } from "@ibrains-design/svelte"
  const [snapshot, send] = useMachine(editable.machine({ id: "1" }))

  const api = $derived(editable.connect(snapshot, send, normalizeProps))
</script>

<div {...api.getRootProps()}>
  <div {...api.getAreaProps()}>
    <input {...api.getInputProps()} />
    <span {...api.getPreviewProps()}></span>
  </div>
  <div>
    {#if !api.editing}
      <button {...api.getEditTriggerProps()}>Edit</button>
    {/if}
    {#if api.editing}
      <div>
        <button {...api.getSubmitTriggerProps()}>Save</button>
        <button {...api.getCancelTriggerProps()}>Cancel</button>
      </div>
    {/if}
  </div>
</div>
```
